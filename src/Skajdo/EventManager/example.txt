<?php

class EventExample extends Event {
	public $test = 'TU MOWI EVENT 1';
}

class SuperEventExample extends Event {
	public $test = 'TU MOWI EVENT 2';
}

class ListenerExample extends Listener {

	public function onEventExample(EventExample $event){
		echo $event->test . ' = TU MOWI LISTENER !' . PHP_EOL;
	}


	public function onSuperEvent(SuperEventExample $event){
		echo $event->test . ' = TU MOWI LISTENER Z SUPER EVENTEM !' . PHP_EOL;
	}
}

class IndexController extends Zend_Controller_Action
{
	public function indexAction()
	{
		$this->getResponse()->setHeader('Content-type', 'text/plain;charset=UTF-8')->sendHeaders();

		$a = new Manager();
		$a->addListener(new ListenerExample());

		$event = new EventExample();
		$a->triggerEvent($event);

		$event = new SuperEventExample();
		$a->triggerEvent($event);

		die;

	}
}
